package closer

import (
        "context"
        "fmt"
)

type (
        Releaser func(context.Context) error
        graph    map[id]map[id]bool
        id       int
)

type Dependency struct {
        id       id
        c        *Closer
        releaser Releaser
}

type Closer struct {
        g      graph
        deps   map[id]*Dependency
        nextId id
}

func (c *Closer) newDependency(r Releaser) *Dependency {
        d := &Dependency{
                id:       c.nextId,
                c:        c,
                releaser: r,
        }
        c.nextId++
        return d
}

// Add registers a dependant resource with releaser for graceful shutdown and associated dependencies.
func (c *Closer) Add(r Releaser, ds ...*Dependency) (*Dependency, error) {
        const op = "instantiating dependency"

        // Verify dependencies.
        for _, d := range ds {
                if d == nil {
                        return nil, fmt.Errorf("%s: nil dependency", op)
                }
                if d.c != c {
                        return nil, fmt.Errorf("%s: dependency is not associated with canceler", op)
                }
                if c.deps[d.id] == nil {
                        return nil, fmt.Errorf("%s: dependency not found", op)
                }
        }

        from := c.newDependency(r)
        err := c.register(from, ds...)
        if err != nil {
                return nil, fmt.Errorf("%s: %v", op, err)
        }
        return from, nil
}

func (c *Closer) register(from *Dependency, ds ...*Dependency) error {
        const op = "register dependencies"

        if c.g == nil {
                c.g = make(graph)
        }
        for _, to := range ds {
                ok := c.g.noCycleWith(from.id, to.id)
                if !ok {
                        return fmt.Errorf("%s: cycle dependency", op)
                }
                end := c.g[from.id]
                if end == nil {
                        end = make(map[id]bool)
                        c.g[from.id] = end
                }
                end[to.id] = true
        }

        if c.deps == nil {
                c.deps = make(map[id]*Dependency)
        }
        c.deps[from.id] = from

        return nil
}

func (g graph) noCycleWith(from id, to id) (ok bool) {
        var (
                visited = make(map[id]bool)
                stack   = make([]id, 0, 1)
                curr    id
        )

        stack = append(stack, to)
        for len(stack) > 0 {
                curr, stack = stack[0], stack[1:]
                if visited[curr] {
                        continue
                }
                visited[curr] = true
                for nodeID := range g[curr] {
                        if nodeID == from {
                                return false // Cycle found.
                        }
                        if visited[nodeID] {
                                continue // Don't visit again.
                        }
                        stack = append(stack, nodeID)
                }
        }

        return true
}
